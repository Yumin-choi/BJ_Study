import java.util.Scanner;

public class Main {

	public static void main(String[] args) {
	Scanner scanner = new Scanner(System.in);
	int n = scanner.nextInt();//3ют╥б╫ц
	for(int i=0; i<n; i++) {	
		for(int j=n-i; j<n; j++) {
			System.out.print(" ");
		}
		for(int k=i; k<n; k++) {
			System.out.print("*");
		}
	
	System.out.println();
	}
}
}
